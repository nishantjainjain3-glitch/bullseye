Get Live NSE/BSE Prices — Free in 15 Minutes
NSE India blocks direct browser requests. A free Python proxy on Render.com fixes this permanently.
Step 1 — Create app.py
from flask import Flask, jsonify
from flask_cors import CORS
import requests

app = Flask(__name__)
CORS(app)
H = {"User-Agent":"Mozilla/5.0","Referer":"https://www.nseindia.com"}

@app.route("/q/<sym>")
def quote(sym):
    s = requests.Session()
    s.get("https://www.nseindia.com", headers=H)
    r = s.get(f"https://www.nseindia.com/api/quote-equity?symbol={sym}", headers=H)
    return jsonify(r.json())

if __name__ == "__main__":
    app.run()
